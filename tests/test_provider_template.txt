# ============================================================================
# AWS Provider Configuration Template for Dual-Mode Testing
# ============================================================================
#
# This template provides reusable provider configuration for tests that work
# with both LocalStack and real AWS.
#
# USAGE:
# Copy this provider block to the top of your .tftest.hcl file, before any
# run blocks. No modifications needed - works out of the box.
#
# RUNNING TESTS:
#   LocalStack: terraform test -var="use_localstack=true"
#         AWS: terraform test -var="use_localstack=false"
#
#   Or use Make:
#   LocalStack: make test-local
#         AWS: make test-aws
#
# COMPATIBILITY:
# Mark your test file with a compatibility header:
#   # LocalStack Compatibility: FULL|PARTIAL|NONE
#   # <Description>
#
# ============================================================================

provider "aws" {
  region = "us-east-1"

  # Skip AWS-specific validations when using LocalStack
  # These would fail in LocalStack as it doesn't implement AWS metadata APIs
  skip_credentials_validation = var.use_localstack
  skip_metadata_api_check     = var.use_localstack
  skip_requesting_account_id  = var.use_localstack

  # CRITICAL: LocalStack requires S3 path-style access
  # Format: s3.amazonaws.com/bucket/key (not bucket.s3.amazonaws.com/key)
  s3_use_path_style = var.use_localstack

  # Dynamic endpoints - only populated when use_localstack = true
  # All services point to LocalStack gateway (http://localhost:4566)
  dynamic "endpoints" {
    for_each = var.use_localstack ? [1] : []
    content {
      apigateway = var.localstack_endpoint
      dynamodb   = var.localstack_endpoint
      events     = var.localstack_endpoint
      iam        = var.localstack_endpoint
      lambda     = var.localstack_endpoint
      route53    = var.localstack_endpoint
      s3         = var.localstack_endpoint
      sns        = var.localstack_endpoint
      sqs        = var.localstack_endpoint
      sts        = var.localstack_endpoint
    }
  }
}
